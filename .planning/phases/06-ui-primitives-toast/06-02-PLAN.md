Create `src/components/ui/ToastProvider.jsx` following the exact `UserContext.jsx` pattern. Per D-08 (React Context-based), D-03 (8s auto-dismiss), D-04 (max 3 toasts), D-05 (error/success/info types).

```jsx
import React, { createContext, useContext, useState, useCallback } from 'react';
import { Toaster } from './Toast';

const ToastContext = createContext();

export const ToastProvider = ({ children }) => {
  const [toasts, setToasts] = useState([]);

  const addToast = useCallback((type, message) => {
    const id = Date.now() + Math.random();
    setToasts(prev => {
      // D-04: max 3 visible — drop oldest if at limit
      const next = prev.length >= 3 ? prev.slice(1) : prev;
      return [...next, { id, type, message, exiting: false }];
    });
  }, []);

  const startExit = useCallback((id) => {
    setToasts(prev => prev.map(t =>
      t.id === id ? { ...t, exiting: true } : t
    ));
    // D-15: fade-out 200ms, then remove from DOM
    setTimeout(() => {
      setToasts(prev => prev.filter(t => t.id !== id));
    }, 200);
  }, []);

  const toast = {
    error: (msg) => addToast('error', msg),
    success: (msg) => addToast('success', msg),
    info: (msg) => addToast('info', msg),
  };

  return (
    <ToastContext.Provider value={toast}>
      {children}
      <Toaster toasts={toasts} onDismiss={startExit} />
    </ToastContext.Provider>
  );
};

export const useToast = () => useContext(ToastContext);
```

Key implementation details:
- `addToast` uses `Date.now() + Math.random()` for unique IDs (same pattern as session IDs in the codebase)
- `startExit` sets `exiting: true` flag then removes after 200ms (per D-15 exit animation duration)
- `toast` object has `.error()`, `.success()`, `.info()` methods matching the API consumers will use
- `Toaster` component renders as a sibling to `children`, NOT inside any router

**Create `src/components/ui/Toast.jsx`** — per D-02 (top-right), D-03 (8s dismiss, X button), D-06 (role="alert" for errors, role="status" for others):

```jsx
import { useEffect } from 'react';
import './Toast.css';

export function ToastItem({ toast, onDismiss }) {
  useEffect(() => {
    if (toast.exiting) return; // Don't set new timer while exiting
    const timer = setTimeout(() => onDismiss(toast.id), 8000);
    return () => clearTimeout(timer);
  }, [toast.id, toast.exiting, onDismiss]);

  return (
    <div
      className={`toast toast--${toast.type}${toast.exiting ? ' toast--exiting' : ''}`}
      role={toast.type === 'error' ? 'alert' : 'status'}
      aria-live={toast.type === 'error' ? 'assertive' : 'polite'}
    >
      <span className="toast__message">{toast.message}</span>
      <button
        className="toast__close"
        onClick={() => onDismiss(toast.id)}
        aria-label="Fechar notificacao"
      >
        ×
      </button>
    </div>
  );
}

export function Toaster({ toasts, onDismiss }) {
  return (
    <div className="toaster" aria-live="polite" aria-relevant="additions removals">
      {toasts.map(t => (
        <ToastItem key={t.id} toast={t} onDismiss={onDismiss} />
      ))}
    </div>
  );
}
```

Key details:
- `Toaster` div is ALWAYS rendered (even when `toasts` is empty) — per Pitfall 1 in RESEARCH.md, `aria-live` container must pre-exist in DOM for NVDA/JAWS compatibility
- `ToastItem` clears its timer in useEffect cleanup (prevents setState on unmounted component)
- `toast.exiting` check prevents setting a new auto-dismiss timer during exit animation
- Close button text is `×` (Unicode multiplication sign) with PT-BR aria-label "Fechar notificacao"
